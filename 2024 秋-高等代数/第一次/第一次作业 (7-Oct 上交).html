    S1=(a11+a21)(b11+b12),S2=(a12+a22)(b21+b22),S3=(a11−a22)(b11+b22),S4=a11(b12−b22),S5=(a21+a22)b11,S6=(a11+a12)b22,S7=a22(b21−b11),
